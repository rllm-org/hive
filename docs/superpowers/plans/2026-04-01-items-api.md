# Items API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add task-scoped items (work items) and comments to the Hive server as REST endpoints.

**Architecture:** Two new tables (`items`, `item_comments`) added to the existing PostgreSQL schema, with 11 new FastAPI endpoints mounted on the existing router under `/tasks/{task_id}/items`. Atomic ID generation via a counter column on the `tasks` table. Soft delete for both items and comments.

**Tech Stack:** Python, FastAPI, psycopg (async), PostgreSQL

**Spec:** `docs/superpowers/specs/2026-04-01-items-api-design.md`

```python
import re
from datetime import datetime, timezone
from typing import Any

from fastapi import APIRouter, HTTPException, Query

from .db import get_db, now, paginate
from .main import JSONResponse, get_agent, _parse_sort

router = APIRouter(prefix="/api/tasks/{task_id}/items")

VALID_STATUSES = {"backlog", "todo", "in_progress", "done", "cancelled"}
VALID_PRIORITIES = {"none", "urgent", "high", "medium", "low"}
_LABEL_RE = re.compile(r"^[a-zA-Z0-9_-]+$")


def _task_prefix(task_id: str) -> str:
    return task_id.split("-")[0].upper()


def _validate_labels(labels: list[str]):
    if len(labels) > 20:
        raise HTTPException(400, "max 20 labels")
    for label in labels:
        if len(label) > 50:
            raise HTTPException(400, f"label too long (max 50): {label}")
        if not _LABEL_RE.match(label):
            raise HTTPException(400, f"label must match [a-zA-Z0-9_-]: {label}")


def _validate_item_fields(body: dict[str, Any]):
    if "title" in body:
        if not body["title"] or len(body["title"]) > 500:
            raise HTTPException(400, "title required and max 500 chars")
    if "description" in body and body["description"] is not None:
        if len(body["description"]) > 10000:
            raise HTTPException(400, "description max 10000 chars")
    if "status" in body and body["status"] not in VALID_STATUSES:
        raise HTTPException(400, f"status must be one of: {', '.join(sorted(VALID_STATUSES))}")
    if "priority" in body and body["priority"] not in VALID_PRIORITIES:
        raise HTTPException(400, f"priority must be one of: {', '.join(sorted(VALID_PRIORITIES))}")
    if "labels" in body:
        _validate_labels(body["labels"])


async def _check_task(task_id: str, conn):
    row = await (await conn.execute("SELECT id FROM tasks WHERE id = %s", (task_id,))).fetchone()
    if not row:
        raise HTTPException(404, "task not found")


async def _get_item(task_id: str, item_id: str, conn) -> dict:
    row = await (await conn.execute(
        "SELECT * FROM items WHERE id = %s AND task_id = %s AND deleted_at IS NULL",
        (item_id, task_id),
    )).fetchone()
    if not row:
        raise HTTPException(404, "item not found")
    return dict(row)


def _item_response(row: dict, comment_count: int = 0) -> dict:
    return {
        "id": row["id"],
        "task_id": row["task_id"],
        "title": row["title"],
        "description": row["description"],
        "status": row["status"],
        "priority": row["priority"],
        "assignee_id": row["assignee_id"],
        "parent_id": row["parent_id"],
        "labels": row["labels"] or [],
        "created_by": row["created_by"],
        "comment_count": comment_count,
        "created_at": row["created_at"],
        "updated_at": row["updated_at"],
    }


async def _comment_count(item_id: str, conn) -> int:
    row = await (await conn.execute(
        "SELECT COUNT(*) AS cnt FROM item_comments WHERE item_id = %s AND deleted_at IS NULL",
        (item_id,),
    )).fetchone()
    return row["cnt"]


@router.post("", status_code=201)
async def create_item(task_id: str, body: dict[str, Any], token: str = Query(...)):
    if "title" not in body or not body.get("title"):
        raise HTTPException(400, "title is required")
    _validate_item_fields(body)
    async with get_db() as conn:
        agent_id = await get_agent(token, conn)
        await _check_task(task_id, conn)
        # Validate parent_id if provided
        parent_id = body.get("parent_id")
        if parent_id:
            parent = await (await conn.execute(
                "SELECT id FROM items WHERE id = %s AND task_id = %s AND deleted_at IS NULL",
                (parent_id, task_id),
            )).fetchone()
            if not parent:
                raise HTTPException(400, "parent_id must reference an existing item in this task")
        # Validate assignee_id if provided
        if body.get("assignee_id"):
            row = await (await conn.execute("SELECT id FROM agents WHERE id = %s", (body["assignee_id"],))).fetchone()
            if not row:
                raise HTTPException(400, "assignee_id must reference an existing agent")
        # Atomic ID generation
        seq_row = await (await conn.execute(
            "UPDATE tasks SET item_seq = item_seq + 1 WHERE id = %s RETURNING item_seq",
            (task_id,),
        )).fetchone()
        seq = seq_row["item_seq"]
        item_id = f"{_task_prefix(task_id)}-{seq}"
        ts = now()
        await conn.execute(
            "INSERT INTO items (id, seq, task_id, title, description, status, priority,"
            " assignee_id, parent_id, labels, created_by, created_at, updated_at)"
            " VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)",
            (
                item_id, seq, task_id,
                body["title"],
                body.get("description"),
                body.get("status", "backlog"),
                body.get("priority", "none"),
                body.get("assignee_id"),
                parent_id,
                body.get("labels", []),
                agent_id, ts, ts,
            ),
        )
        row = await _get_item(task_id, item_id, conn)
    return JSONResponse(_item_response(row, 0), status_code=201)


@router.get("/{item_id}")
async def get_item(task_id: str, item_id: str, token: str = Query(...)):
    async with get_db() as conn:
        await get_agent(token, conn)
        await _check_task(task_id, conn)
        row = await _get_item(task_id, item_id, conn)
        cc = await _comment_count(item_id, conn)
        children_rows = await (await conn.execute(
            "SELECT id, title, status FROM items"
            " WHERE parent_id = %s AND task_id = %s AND deleted_at IS NULL"
            " ORDER BY seq",
            (item_id, task_id),
        )).fetchall()
        resp = _item_response(row, cc)
        resp["children"] = [{"id": c["id"], "title": c["title"], "status": c["status"]} for c in children_rows]
    return JSONResponse(resp)
```

_SORT_KEYS = {
    "recent": "i.created_at",
    "updated": "i.updated_at",
    "priority": "CASE i.priority WHEN 'urgent' THEN 0 WHEN 'high' THEN 1 WHEN 'medium' THEN 2 WHEN 'low' THEN 3 ELSE 4 END",
}


@router.get("")
async def list_items(
    task_id: str,
    token: str = Query(...),
    status: str | None = Query(None),
    priority: str | None = Query(None),
    assignee: str | None = Query(None),
    label: str | None = Query(None),
    parent: str | None = Query(None),
    sort: str = Query("recent"),
    page: int = Query(1),
    per_page: int = Query(20),
):
    page, per_page, offset = paginate(page, per_page)
    async with get_db() as conn:
        await get_agent(token, conn)
        await _check_task(task_id, conn)
        where = ["i.task_id = %s", "i.deleted_at IS NULL"]
        params: list = [task_id]
        if status:
            if status.startswith("!"):
                where.append("i.status != %s")
                params.append(status[1:])
            else:
                where.append("i.status = %s")
                params.append(status)
        if priority:
            where.append("i.priority = %s")
            params.append(priority)
        if assignee:
            if assignee == "none":
                where.append("i.assignee_id IS NULL")
            else:
                where.append("i.assignee_id = %s")
                params.append(assignee)
        if label:
            where.append("i.labels @> %s::text[]")
            params.append([label])
        if parent:
            where.append("i.parent_id = %s")
            params.append(parent)
        order = _parse_sort(sort, _SORT_KEYS)
        # For priority sort, default direction should be ASC (urgent first)
        if sort == "priority":
            order = _parse_sort("priority:asc", _SORT_KEYS)
        where_sql = " AND ".join(where)
        params.extend([per_page + 1, offset])
        rows = await (await conn.execute(
            f"SELECT i.*, (SELECT COUNT(*) FROM item_comments c"
            f" WHERE c.item_id = i.id AND c.deleted_at IS NULL) AS comment_count"

async def _check_cycle(item_id: str, new_parent_id: str, conn):
    """Walk parent chain from new_parent_id. Reject if item_id is found or depth > 5."""
    if new_parent_id == item_id:
        raise HTTPException(400, "cycle detected: item cannot be its own parent")
    current = new_parent_id
    depth = 0
    while current is not None and depth < 5:
        if current == item_id:
            raise HTTPException(400, "cycle detected: would create circular parent chain")
        row = await (await conn.execute(
            "SELECT parent_id FROM items WHERE id = %s AND deleted_at IS NULL", (current,)
        )).fetchone()
        current = row["parent_id"] if row else None
        depth += 1
    if depth >= 5:
        raise HTTPException(400, "max depth of 5 exceeded")


_UPDATABLE_FIELDS = {"title", "description", "status", "priority", "assignee_id", "parent_id", "labels"}


@router.patch("/{item_id}")
async def patch_item(task_id: str, item_id: str, body: dict[str, Any], token: str = Query(...)):
    updates = {k: v for k, v in body.items() if k in _UPDATABLE_FIELDS}
    if not updates:
        raise HTTPException(400, "nothing to update")
    _validate_item_fields(updates)
    async with get_db() as conn:
        await get_agent(token, conn)
        await _check_task(task_id, conn)
        item = await _get_item(task_id, item_id, conn)
        if "parent_id" in updates and updates["parent_id"] is not None:
            parent = await (await conn.execute(
                "SELECT id FROM items WHERE id = %s AND task_id = %s AND deleted_at IS NULL",
                (updates["parent_id"], task_id),
            )).fetchone()
            if not parent:
                raise HTTPException(400, "parent_id must reference an existing item in this task")
            await _check_cycle(item_id, updates["parent_id"], conn)
        if "assignee_id" in updates and updates["assignee_id"] is not None:
            row = await (await conn.execute("SELECT id FROM agents WHERE id = %s", (updates["assignee_id"],))).fetchone()
            if not row:
                raise HTTPException(400, "assignee_id must reference an existing agent")
        sets = ", ".join(f"{k} = %s" for k in updates)
        vals = list(updates.values())
        vals.append(now())
        vals.append(item_id)
        vals.append(task_id)
        await conn.execute(
            f"UPDATE items SET {sets}, updated_at = %s WHERE id = %s AND task_id = %s",
            vals,
        )
        row = await _get_item(task_id, item_id, conn)
        cc = await _comment_count(item_id, conn)
    return JSONResponse(_item_response(row, cc))

@router.get("/{item_id}/comments")
async def list_comments(
    task_id: str, item_id: str,
    token: str = Query(...),
    page: int = Query(1),
    per_page: int = Query(30),
):
    page, per_page, offset = paginate(page, per_page)
    async with get_db() as conn:
        await get_agent(token, conn)
        await _check_task(task_id, conn)
        await _get_item(task_id, item_id, conn)
        rows = await (await conn.execute(
            "SELECT id, item_id, agent_id, content, created_at"
            " FROM item_comments WHERE item_id = %s AND deleted_at IS NULL"
            " ORDER BY created_at ASC LIMIT %s OFFSET %s",
            (item_id, per_page + 1, offset),
        )).fetchall()
        has_next = len(rows) > per_page
        rows = rows[:per_page]
    return JSONResponse({
        "comments": [dict(r) for r in rows],
        "page": page, "per_page": per_page, "has_next": has_next,
    })
